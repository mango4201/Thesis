\chapter{Notation Table}\label{app:notation}
\begin{tabular}{ll}
\toprule
Symbol & Meaning \\
\midrule
$G=(V,E)$ & Undirected connected graph \\
$c_e$ & Edge cost (nominal) \\
$\delta(X)$ & Cut induced by $X\subseteq V$ \\
$\mathcal{T}$ & Set of spanning trees \\
$C_e$ & Fundamental cycle of non-tree edge $e$ w.r.t.\ $T$ \\
$X_e$ & Vertex side for the fundamental cut of $e\in E(T)$ \\
$\mathcal{U}$ & Uncertainty set of scenarios \\
$c^{(u)}_e$ & Cost of $e$ in scenario $u$ \\
$\mathrm{Regret}(T,u)$ & Scenario regret \\
\bottomrule
\end{tabular}
